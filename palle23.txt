BASE_URL = "https://api.brawlstars.com/v1"
SOGLIA = 1000          # coppe target
INTERVALLO = 60        # secondi tra ogni controllo


def format_tag(tag: str) -> str:
    tag = tag.strip().upper()
    if not tag.startswith("#"):
        tag = "#" + tag
    return tag


def get_player(tag: str) -> dict | None:
    encoded_tag = requests.utils.quote(tag, safe="")
    url = f"{BASE_URL}/players/{encoded_tag}"
    headers = {"Authorization": f"Bearer {API_KEY}", "Accept": "application/json"}
    try:
        r = requests.get(url, headers=headers, timeout=10)
        if r.status_code == 200:
            return r.json()
        else:
            print(f"  ⚠️  Errore API {r.status_code}, riprovo al prossimo ciclo...")
            return None
    except Exception as e:
        print(f"  ⚠️  Errore di rete: {e}")
        return None


def get_brawler_trophies(player_data: dict, brawler_name: str) -> int | None:
    search = brawler_name.strip().upper()
    for b in player_data.get("brawlers", []):
        if b["name"].upper() == search:
            return b.get("trophies", 0)
    return None


def send_discord_notification(player_name: str, brawler_name: str, trophies: int):
    """Invia 5 messaggi al webhook pingando @everyone."""
    messaggio = (
        f"@everyone 🏆 **{player_name}** ha raggiunto **{trophies} coppe** "
        f"con il brawler **{brawler_name}**! 🎉"
    )
    print(f"\n🔔 Invio notifica Discord (5 volte)...")
    for i in range(1, 6):
        payload = {"content": messaggio}
        try:
            r = requests.post(WEBHOOK_URL, json=payload, timeout=10)
            if r.status_code in (200, 204):
                print(f"  ✅ Messaggio {i}/5 inviato")
            else:
                print(f"  ❌ Messaggio {i}/5 fallito ({r.status_code}): {r.text}")
        except Exception as e:
            print(f"  ❌ Errore invio {i}/5: {e}")
        if i < 5:
            time.sleep(1.5)  # piccola pausa tra i messaggi


def list_brawlers(player_data: dict):
    tutti = sorted(b["name"].capitalize() for b in player_data.get("brawlers", []))
    print(f"\n📋 Brawler posseduti ({len(tutti)}):")
    for i in range(0, len(tutti), 5):
        riga = "   ".join(f"{b:<14}" for b in tutti[i:i+5])
        print(f"   {riga}")


def main():
    print("=" * 56)
    print("  🌟  Brawl Stars — Monitor Coppe → Discord")
    print("=" * 56)

    # Input tag
    tag_input = input("\n🏷️  Tag del giocatore (es. #ABC123): ").strip()
    tag = format_tag(tag_input)

    # Carica giocatore per mostrare brawler disponibili
    print(f"\n🔍 Carico dati per {tag}...")
    player = get_player(tag)
    if not player:
        print("❌ Impossibile caricare il giocatore. Controlla il tag.\n")
        sys.exit(1)

    player_name = player.get("name", "Sconosciuto")
    print(f"✅ Giocatore: {player_name}")
    list_brawlers(player)

    # Input brawler
    brawler_input = input("\n🎮 Nome del brawler da monitorare: ").strip().upper()
    if not brawler_input:
        print("❌ Nome non valido.\n")
        sys.exit(1)

    # Verifica che il brawler esista
    trophies_now = get_brawler_trophies(player, brawler_input)
    if trophies_now is None:
        print(f"❌ Brawler '{brawler_input}' non trovato. Controlla il nome.\n")
        sys.exit(1)

    print(f"\n✅ Brawler trovato: {brawler_input.capitalize()}")
    print(f"   Coppe attuali : {trophies_now:,}")
    print(f"   Soglia target : {SOGLIA:,}")
    print(f"   Controllo ogni: {INTERVALLO}s")

    if trophies_now >= SOGLIA:
        print(f"\n⚡ Soglia già raggiunta! Invio notifica subito...")
        send_discord_notification(player_name, brawler_input.capitalize(), trophies_now)
        print("\n✅ Fatto!\n")
        return

    print(f"\n🔄 Monitoraggio avviato... (Ctrl+C per fermare)\n")
    notifica_inviata = False

    while not notifica_inviata:
        player_data = get_player(tag)
        if player_data:
            trophies = get_brawler_trophies(player_data, brawler_input)
            if trophies is not None:
                ts = time.strftime("%H:%M:%S")
                print(f"  [{ts}] {brawler_input.capitalize()}: {trophies:,} coppe  (target: {SOGLIA:,})")

                if trophies >= SOGLIA:
                    send_discord_notification(
                        player_data.get("name", player_name),
                        brawler_input.capitalize(),
                        trophies
                    )
                    notifica_inviata = True
                    print("\n✅ Notifica inviata! Monitoraggio terminato.\n")
                    break
            else:
                print(f"  ⚠️  Brawler non trovato nella risposta.")

        time.sleep(INTERVALLO)


if __name__ == "__main__":
    try:
        main()
    except KeyboardInterrupt:
        print("\n\n⛔ Monitoraggio fermato dall'utente.\n")
